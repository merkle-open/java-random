5   package com.namics.commons.random.generator;
6   
7   /**
8    * Interface to be Implemented to add a custom RandomGenerator for a custom SupportedType.
9    * The generator can be registered in several ways:
10   * <li>automatically when placed in same package as the SupportedType</li>
11   * <li>Manually by registering either class or instance directly to {@link com.namics.commons.random.RandomData#addRandomGenerator}</li>
12   * <li>Manually by registering package for scanning  to {@link com.namics.commons.random.RandomData#addRandomGenerators(String)}</li>
13   *
14   * @param <SupportedType> Type that can be created with this Generator.
15   */
16  public interface RandomGenerator<SupportedType> {
17  
18  	/**
19  	 * Method that creates the random object of SupportedType.
20  	 *
21  	 * @return Random object of SupportedType.
22  	 */
23  	public SupportedType random();
24  
25  	/**
26  	 * Method must return the actual classes of SupportedType (SupportedType.class).
27  	 *
28  	 * @return Types supported by this generator implementation.
29  	 */
30  	public Iterable<Class<SupportedType>> supportedTypes();
31  }

7   import com.namics.commons.random.generator.RandomGenerator;

9   import java.util.Arrays;
10  import java.util.Calendar;

12  /**
13   * CalendarGenerator.
14   *
15   * @author sbaur
16   * @since 03.03.15 08:58
17   */
18  public class CalendarGenerator implements RandomGenerator<Calendar>
19      {
20  
21          @Override
22          public Calendar random()
23          {
24              Calendar calendar = Calendar.getInstance();
25              calendar.setTime(new DateGenerator().random());
26              return calendar;
27          }
28  
29          @Override
30          public Iterable<Class<Calendar>> supportedTypes()
31          {
32              return Arrays.asList(Calendar.class);
33          }
34  
35      }
